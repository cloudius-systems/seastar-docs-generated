123 template<uint64_t Min, uint64_t Max, size_t Precision>
124 class approximate_exponential_histogram {
125  static constexpr unsigned NUM_EXP_RANGES = log2floor(Max/Min);
126  static constexpr size_t NUM_BUCKETS = NUM_EXP_RANGES * Precision + 1;
127  static constexpr unsigned PRECISION_BITS = log2floor(Precision);
128  static constexpr unsigned BASESHIFT = log2floor(Min);
129  static constexpr uint64_t LOWER_BITS_MASK = Precision - 1;
130  std::array<uint64_t, NUM_BUCKETS> _buckets;
131 public:
132  approximate_exponential_histogram() {
133  clear();
134  }
135 
141  uint64_t get_bucket_lower_limit(uint16_t bucket_id) const {
142  if (bucket_id == NUM_BUCKETS - 1) {
143  return Max;
144  }
145  int16_t exp_rang = (bucket_id >> PRECISION_BITS);
146  return (Min << exp_rang) + ((bucket_id & LOWER_BITS_MASK) << (exp_rang + BASESHIFT - PRECISION_BITS));
147  }
148 
154  uint64_t get_bucket_upper_limit(uint16_t bucket_id) const {
155  if (bucket_id == NUM_BUCKETS - 1) {
156  return std::numeric_limits<uint64_t>::max();
157  }
158  return get_bucket_lower_limit(bucket_id + 1);
159  }
160 
166  uint16_t find_bucket_index(uint64_t val) const {
167  if (val >= Max) {
168  return NUM_BUCKETS - 1;
169  }
170  if (val <= Min) {
171  return 0;
172  }
173  uint16_t range = log2floor(val);
174  val >>= range - PRECISION_BITS; // leave the top most N+1 bits where N is the resolution.
175  return ((range - BASESHIFT) << PRECISION_BITS) + (val & LOWER_BITS_MASK);
176  }

181  void clear() {
182  std::fill(_buckets.begin(), _buckets.end(), 0);
183  }
184 
189  void add(uint64_t n) {
190  _buckets.at(find_bucket_index(n))++;
191  }

200  uint64_t min() const {
201  for (size_t i = 0; i < NUM_BUCKETS; i ++) {
202  if (_buckets[i] > 0) {
203  return get_bucket_lower_limit(i);
204  }
205  }
206  return 0;
207  }

216  uint64_t max() const {
217  for (int i = NUM_BUCKETS - 1; i >= 0; i--) {
218  if (_buckets[i] > 0) {

221  }

223  }

260  uint64_t quantile(float quantile) const {
261  if (quantile < 0 || quantile > 1.0) {
262  throw std::runtime_error("Invalid quantile value " + std::to_string(quantile) + ". Value should be between 0 and 1");
263  }
264  auto c = count();
265 
266  if (!c) {
267  return 0; // no data
268  }
269 
270  auto pcount = uint64_t(std::floor(c * quantile));
271  uint64_t elements = 0;
272  for (size_t i = 0; i < NUM_BUCKETS - 2; i++) {
273  if (_buckets[i]) {
274  elements += _buckets[i];
275  if (elements >= pcount) {
276  return get_bucket_lower_limit(i);
277  }
278  }
279  }
280  return Max; // overflowed value is in the requested quantile
281  }

287  uint64_t mean() const {
288  uint64_t elements = 0;
289  double sum = 0;
290  for (size_t i = 0; i < NUM_BUCKETS - 1; i++) {
291  elements += _buckets[i];
292  sum += _buckets[i] * get_bucket_lower_limit(i);
293  }
294  return (elements) ? sum / elements : 0;
295  }

329  seastar::metrics::histogram to_metrics_histogram() const noexcept {
330  seastar::metrics::histogram res;
331  res.buckets.resize(size() - 1);
332  uint64_t cummulative_count = 0;
333  res.sample_sum = 0;
334 
335  for (size_t i = 0; i < NUM_BUCKETS - 1; i++) {
336  auto& v = res.buckets[i];
337  v.upper_bound = get_bucket_lower_limit(i + 1);
338  cummulative_count += get(i);
339  v.count = cummulative_count;
340  res.sample_sum += get(i) * v.upper_bound;
341  }
342  // The count serves as the infinite bucket
343  res.sample_count = cummulative_count + get(NUM_BUCKETS - 1);
344  res.sample_sum += get(NUM_BUCKETS - 1) * get_bucket_lower_limit(NUM_BUCKETS - 1);
345  return res;
346  }
