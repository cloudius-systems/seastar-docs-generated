68 template<typename T>
69 T* scheduling_group_get_specific_ptr(scheduling_group sg, scheduling_group_key key) noexcept {
70  auto& data = internal::get_scheduling_group_specific_thread_local_data();
71 #ifdef SEASTAR_DEBUG
72  assert(std::type_index(typeid(T)) == data.scheduling_group_key_configs[key.id()].type_index);
73 #endif
74  auto sg_id = internal::scheduling_group_index(sg);
75  if (__builtin_expect(sg_id < data.per_scheduling_group_data.size() &&
76  data.per_scheduling_group_data[sg_id].queue_is_initialized, true)) {
77  return reinterpret_cast<T*>(data.per_scheduling_group_data[sg_id].specific_vals[key.id()]);
78  }
79  return nullptr;
80 }

109 template<typename T>
110 T& scheduling_group_get_specific(scheduling_group_key key) noexcept {
111  // Unlike internal::scheduling_group_get_specific_ptr, this can

113  // false.
114  auto& data = internal::get_scheduling_group_specific_thread_local_data();
