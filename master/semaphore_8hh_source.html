103 template<typename ExceptionFactory, typename Clock = typename timer<>::clock>
104 class basic_semaphore : private ExceptionFactory {
105 public:
106  using duration = typename timer<Clock>::duration;
107  using clock = typename timer<Clock>::clock;
108  using time_point = typename timer<Clock>::time_point;
109  using exception_factory = ExceptionFactory;
110 private:
111  ssize_t _count;
112  std::exception_ptr _ex;
113  struct entry {
114  promise<> pr;
115  size_t nr;
116  entry(promise<>&& pr_, size_t nr_) noexcept : pr(std::move(pr_)), nr(nr_) {}
117  };
118  using expiry_handler = std::function<void (entry&)>;
119  expiring_fifo<entry, expiry_handler, clock> _wait_list;
120  expiry_handler make_expiry_handler() noexcept {
121  return [this] (entry& e) noexcept {
122  try {
123  e.pr.set_exception(this->timeout());
124  } catch (...) {
125  e.pr.set_exception(semaphore_timed_out());
126  }
127  };
128  }
129  bool has_available_units(size_t nr) const noexcept {
130  return _count >= 0 && (static_cast<size_t>(_count) >= nr);
131  }
132  bool may_proceed(size_t nr) const noexcept {
133  return has_available_units(nr) && _wait_list.empty();
134  }
135 public:
137  static constexpr size_t max_counter() noexcept {
138  return std::numeric_limits<decltype(_count)>::max();
139  }

146  basic_semaphore(size_t count) noexcept(std::is_nothrow_default_constructible_v<exception_factory>)
147  : exception_factory()
148  , _count(count),
149  _wait_list(make_expiry_handler())
150  {}
151  basic_semaphore(size_t count, exception_factory&& factory) noexcept(std::is_nothrow_move_constructible_v<exception_factory>)
152  : exception_factory(std::move(factory))
153  , _count(count)
154  , _wait_list(make_expiry_handler())
155  {
156  static_assert(std::is_nothrow_move_constructible_v<expiry_handler>);
157  }
168  future<> wait(size_t nr = 1) noexcept {
169  return wait(time_point::max(), nr);
170  }
184  future<> wait(time_point timeout, size_t nr = 1) noexcept {
185  if (may_proceed(nr)) {
186  _count -= nr;
187  return make_ready_future<>();
188  }
189  if (_ex) {
190  return make_exception_future(_ex);
191  }
192  entry e(promise<>(), nr);
193  auto fut = e.pr.get_future();
194  try {
195  _wait_list.push_back(std::move(e), timeout);
196  } catch (...) {
197  e.pr.set_exception(std::current_exception());
198  }
199  return fut;
200  }

215  future<> wait(duration timeout, size_t nr = 1) noexcept {
216  return wait(clock::now() + timeout, nr);
217  }
227  void signal(size_t nr = 1) noexcept {
228  if (_ex) {
229  return;
230  }
231  _count += nr;
232  while (!_wait_list.empty() && has_available_units(_wait_list.front().nr)) {
233  auto& x = _wait_list.front();
234  _count -= x.nr;
235  x.pr.set_value();
236  _wait_list.pop_front();
237  }
238  }

332 template<typename ExceptionFactory = semaphore_default_exception_factory, typename Clock = typename timer<>::clock>
333 class semaphore_units {
334  basic_semaphore<ExceptionFactory, Clock>* _sem;
335  size_t _n;
336 
337  semaphore_units(basic_semaphore<ExceptionFactory, Clock>* sem, size_t n) noexcept : _sem(sem), _n(n) {}
338 public:
339  semaphore_units() noexcept : semaphore_units(nullptr, 0) {}
340  semaphore_units(basic_semaphore<ExceptionFactory, Clock>& sem, size_t n) noexcept : semaphore_units(&sem, n) {}
341  semaphore_units(semaphore_units&& o) noexcept : _sem(o._sem), _n(std::exchange(o._n, 0)) {
342  }
343  semaphore_units& operator=(semaphore_units&& o) noexcept {
344  _sem = o._sem;
345  _n = std::exchange(o._n, 0);
346  return *this;
347  }
348  semaphore_units(const semaphore_units&) = delete;
349  ~semaphore_units() noexcept {
350  return_all();
351  }
359  size_t return_units(size_t units) {
360  if (units > _n) {
361  throw std::invalid_argument("Cannot take more units than those protected by the semaphore");
362  }
363  _n -= units;
364  _sem->signal(units);
365  return _n;
366  }
368  void return_all() noexcept {
369  if (_n) {
370  _sem->signal(_n);
371  _n = 0;
372  }
373  }
377  size_t release() noexcept {
378  return std::exchange(_n, 0);
379  }
388  semaphore_units split(size_t units) {
389  if (units > _n) {
390  throw std::invalid_argument("Cannot take more units than those protected by the semaphore");
391  }
392  _n -= units;
393  return semaphore_units(_sem, units);
394  }
400  void adopt(semaphore_units&& other) noexcept {
401  assert(other._sem == _sem);
402  _n += other.release();
403  }
404 
406  size_t count() const noexcept {
407  return _n;
408  }
