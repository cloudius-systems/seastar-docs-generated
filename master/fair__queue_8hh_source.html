334  clock_type next_pending_aio() const noexcept {
335  if (_pending) {
336  /*
337  * We expect the disk to release the ticket within some time,
338  * but it's ... OK if it doesn't -- the pending wait still
339  * needs the head rover value to be ahead of the needed value.
340  *
341  * It may happen that the capacity gets released before we think
342  * it will, in this case we will wait for the full value again,
343  * which's sub-optimal. The expectation is that we think disk
344  * works faster, than it really does.
345  */
346  fair_queue_ticket over = _pending->head.maybe_ahead_of(_group.head());
347  return std::chrono::steady_clock::now() + duration(over);
348  }
349 
350  return std::chrono::steady_clock::time_point::max();
351  }
