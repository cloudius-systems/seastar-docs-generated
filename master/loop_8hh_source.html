48 template <typename AsyncAction>
49 class repeater final : public continuation_base<stop_iteration> {
50  promise<> _promise;
51  AsyncAction _action;
52 public:
53  explicit repeater(AsyncAction&& action) : _action(std::move(action)) {}
54  future<> get_future() { return _promise.get_future(); }
55  task* waiting_task() noexcept override { return _promise.waiting_task(); }
56  virtual void run_and_dispose() noexcept override {
57  if (_state.failed()) {
58  _promise.set_exception(std::move(_state).get_exception());
59  delete this;
60  return;
61  } else {
62  if (_state.get0() == stop_iteration::yes) {
63  _promise.set_value();
64  delete this;
65  return;
66  }
67  _state = {};
68  }
69  try {
70  do {
71  auto f = futurize_invoke(_action);
72  if (!f.available()) {
73  internal::set_callback(f, this);
74  return;
75  }
76  if (f.get0() == stop_iteration::yes) {
77  _promise.set_value();
78  delete this;
79  return;
80  }
81  } while (!need_preempt());
82  } catch (...) {
83  _promise.set_exception(std::current_exception());
84  delete this;
85  return;
86  }
87  _state.set(stop_iteration::no);
88  schedule(this);
89  }
90 };

110 template<typename AsyncAction>
111 SEASTAR_CONCEPT( requires seastar::InvokeReturns<AsyncAction, stop_iteration> || seastar::InvokeReturns<AsyncAction, future<stop_iteration>> )
112 inline
113 future<> repeat(AsyncAction&& action) noexcept {
114  using futurator = futurize<std::result_of_t<AsyncAction()>>;
115  static_assert(std::is_same<future<stop_iteration>, typename futurator::type>::value, "bad AsyncAction signature");
116  for (;;) {
117  // Do not type-erase here in case this is a short repeat()
118  auto f = futurator::invoke(action);
119 
120  if (!f.available() || f.failed() || need_preempt()) {
121  return [&] () noexcept {
122  memory::scoped_critical_alloc_section _;
123  auto repeater = new internal::repeater<AsyncAction>(std::move(action));
124  auto ret = repeater->get_future();
125  internal::set_callback(f, repeater);
126  return ret;
127  }();
128  }
129 
130  if (f.get0() == stop_iteration::yes) {
131  return make_ready_future<>();
132  }
133  }
134 }

162 class repeat_until_value_state final : public continuation_base<std::optional<T>> {
163  promise<T> _promise;

165 public:

197  return;

199  } while (!need_preempt());
200  } catch (...) {
201  _promise.set_exception(std::current_exception());
202  delete this;

228 repeat_until_value_return_type<AsyncAction>
229 repeat_until_value(AsyncAction action) noexcept {
230  using futurator = futurize<std::result_of_t<AsyncAction()>>;
231  using type_helper = repeat_until_value_type_helper<typename futurator::type>;
232  // the "T" in the documentation
233  using value_type = typename type_helper::value_type;
234  using optional_type = typename type_helper::optional_type;
235  do {
236  auto f = futurator::invoke(action);
237 
238  if (!f.available()) {
239  return [&] () noexcept {
240  memory::scoped_critical_alloc_section _;
241  auto state = new internal::repeat_until_value_state<AsyncAction, value_type>(std::move(action));
242  auto ret = state->get_future();
243  internal::set_callback(f, state);
244  return ret;
245  }();
246  }
247 
248  if (f.failed()) {
249  return make_exception_future<value_type>(f.get_exception());
250  }
251 
252  optional_type&& optional = std::move(f).get0();
253  if (optional) {
254  return make_ready_future<value_type>(std::move(optional.value()));
255  }
256  } while (!need_preempt());
257 
258  try {
259  auto state = new internal::repeat_until_value_state<AsyncAction, value_type>(std::nullopt, std::move(action));
260  auto f = state->get_future();
261  schedule(state);
262  return f;
263  } catch (...) {
264  return make_exception_future<value_type>(std::current_exception());
265  }

270 template <typename StopCondition, typename AsyncAction>
271 class do_until_state final : public continuation_base<> {
272  promise<> _promise;
273  StopCondition _stop;
274  AsyncAction _action;
275 public:
276  explicit do_until_state(StopCondition stop, AsyncAction action) : _stop(std::move(stop)), _action(std::move(action)) {}
277  future<> get_future() { return _promise.get_future(); }
278  task* waiting_task() noexcept override { return _promise.waiting_task(); }
279  virtual void run_and_dispose() noexcept override {
280  if (_state.available()) {
281  if (_state.failed()) {
282  _promise.set_urgent_state(std::move(_state));
283  delete this;
284  return;
285  }
286  _state = {}; // allow next cycle to overrun state
287  }
288  try {
289  do {
290  if (_stop()) {
291  _promise.set_value();
292  delete this;
293  return;
294  }
295  auto f = _action();
296  if (!f.available()) {
297  internal::set_callback(f, this);
298  return;
299  }
300  if (f.failed()) {
301  f.forward_to(std::move(_promise));
302  delete this;
303  return;
304  }
305  } while (!need_preempt());
306  } catch (...) {
307  _promise.set_exception(std::current_exception());
308  delete this;
309  return;
310  }
311  schedule(this);
312  }
313 };

329 inline
330 future<> do_until(StopCondition stop_cond, AsyncAction action) noexcept {
331  using namespace internal;
332  for (;;) {
333  if (stop_cond()) {
334  return make_ready_future<>();
335  }
336  auto f = futurize_invoke(action);
337  if (f.failed()) {
338  return f;
339  }

342  memory::scoped_critical_alloc_section _;
343  auto task = new do_until_state<StopCondition, AsyncAction>(std::move(stop_cond), std::move(action));
344  auto ret = task->get_future();
345  internal::set_callback(f, task);
346  return ret;
347  }();
348  }

371 template <typename Iterator, typename AsyncAction>
372 class do_for_each_state final : public continuation_base<> {
373  Iterator _begin;
374  Iterator _end;
375  AsyncAction _action;
376  promise<> _pr;
377 
378 public:
379  do_for_each_state(Iterator begin, Iterator end, AsyncAction action, future<> first_unavailable)
380  : _begin(std::move(begin)), _end(std::move(end)), _action(std::move(action)) {
381  internal::set_callback(first_unavailable, this);
382  }
383  virtual void run_and_dispose() noexcept override {
384  std::unique_ptr<do_for_each_state> zis(this);
385  if (_state.failed()) {
386  _pr.set_urgent_state(std::move(_state));
387  return;
388  }
389  while (_begin != _end) {
390  auto f = futurize_invoke(_action, *_begin++);
391  if (f.failed()) {
392  f.forward_to(std::move(_pr));
393  return;
394  }
395  if (!f.available() || need_preempt()) {
396  _state = {};
397  internal::set_callback(f, this);
398  zis.release();
399  return;
400  }
401  }
402  _pr.set_value();
403  }
404  task* waiting_task() noexcept override {
405  return _pr.waiting_task();
406  }
407  future<> get_future() {
408  return _pr.get_future();
409  }
410 };

414 future<> do_for_each_impl(Iterator begin, Iterator end, AsyncAction action) {
415  while (begin != end) {
416  auto f = futurize_invoke(action, *begin++);

538 template <typename Iterator, typename Func>
539 SEASTAR_CONCEPT( requires requires (Func f, Iterator i) { { f(*i++) } -> std::same_as<future<>>; } )
540 inline
541 future<>
542 parallel_for_each(Iterator begin, Iterator end, Func&& func) noexcept {
543  parallel_for_each_state* s = nullptr;
544  // Process all elements, giving each future the following treatment:
545  // - available, not failed: do nothing
546  // - available, failed: collect exception in ex
547  // - not available: collect in s (allocating it if needed)
548  while (begin != end) {
549  auto f = futurize_invoke(std::forward<Func>(func), *begin++);
550  if (!f.available() || f.failed()) {
551  if (!s) {
552  memory::scoped_critical_alloc_section _;
553  using itraits = std::iterator_traits<Iterator>;
554  auto n = (internal::iterator_range_estimate_vector_capacity(begin, end, typename itraits::iterator_category()) + 1);
555  s = new parallel_for_each_state(n);
556  }
557  s->add_future(std::move(f));
558  }
559  }
560  // If any futures were not available, hand off to parallel_for_each_state::start().
561  // Otherwise we can return a result immediately.
562  if (s) {
563  // s->get_future() takes ownership of s (and chains it to one of the futures it contains)
564  // so this isn't a leak
565  return s->get_future();
566  }
567  return make_ready_future<>();
568 }

627 SEASTAR_CONCEPT( requires requires (Func f, Iterator i) { { f(*i++) } -> std::same_as<future<>>; } )
628 inline
629 future<>
630 max_concurrent_for_each(Iterator begin, Iterator end, size_t max_concurrent, Func&& func) noexcept {
631  struct state {
632  Iterator begin;
633  Iterator end;
634  Func func;
635  size_t max_concurrent;
636  semaphore sem;
637  std::exception_ptr err;
638 
639  state(Iterator begin_, Iterator end_, size_t max_concurrent_, Func func_)
640  : begin(std::move(begin_))
641  , end(std::move(end_))
642  , func(std::move(func_))
643  , max_concurrent(max_concurrent_)
644  , sem(max_concurrent_)
645  , err()
646  { }
647  };
648 
649  assert(max_concurrent > 0);
650 
651  try {
652  return do_with(state(std::move(begin), std::move(end), max_concurrent, std::forward<Func>(func)), [] (state& s) {
653  return do_until([&s] { return s.begin == s.end; }, [&s] {
654  return s.sem.wait().then([&s] () mutable noexcept {
655  // Possibly run in background and signal _sem when the task is done.
656  // The background tasks are waited on using _sem.
657  (void)futurize_invoke(s.func, *s.begin++).then_wrapped([&s] (future<> fut) {
658  if (fut.failed()) {
659  auto e = fut.get_exception();;
660  if (!s.err) {
661  s.err = std::move(e);
662  }
663  }
664  s.sem.signal();
665  });
666  });
667  }).then([&s] {
668  // Wait for any background task to finish
669  // and signal and semaphore
670  return s.sem.wait(s.max_concurrent);
671  }).then([&s] {
672  if (!s.err) {
673  return make_ready_future<>();
674  }
675  return seastar::make_exception_future<>(std::move(s.err));
676  });
677  });
678  } catch (...) {
679  return current_exception_as_future();
680  }
681 }
