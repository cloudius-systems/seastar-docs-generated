157 // Helper for measuring time.
158 // Each microbenchmark can either use the default behaviour which measures
159 // only the start and stop time of the whole run or manually invoke
160 // start_measuring_time() and stop_measuring_time() in order to measure
161 // only parts of each iteration.
162 class time_measurement {
163  clock_type::time_point _run_start_time;
164  clock_type::time_point _start_time;
165  clock_type::duration _total_time;
166 
167  perf_stats _start_stats;
168  perf_stats _total_stats;
169 
170  linux_perf_event* _instructions_retired_counter = nullptr;
171 
172 public:
173  [[gnu::always_inline]] [[gnu::hot]]
174  void start_run(linux_perf_event* instructions_retired_counter = nullptr) {
175  _instructions_retired_counter = instructions_retired_counter;
176  _total_time = { };
177  _total_stats = {};
178  auto t = clock_type::now();
179  _run_start_time = t;
180  _start_time = t;
181  _start_stats = perf_stats::snapshot(_instructions_retired_counter);
182  }
183 
184  [[gnu::always_inline]] [[gnu::hot]]
185  performance_test::run_result stop_run() {
186  auto t = clock_type::now();
187  performance_test::run_result ret;
188  if (_start_time == _run_start_time) {
189  ret.duration = t - _start_time;
190  auto stats = perf_stats::snapshot(_instructions_retired_counter);
191  ret.stats = stats - _start_stats;
192  } else {
193  ret.duration = _total_time;
194  ret.stats = _total_stats;
195  }
196  _instructions_retired_counter = nullptr;
197  return ret;
198  }
199 
200  [[gnu::always_inline]] [[gnu::hot]]
201  void start_iteration() {
202  _start_time = clock_type::now();
203  _start_stats = perf_stats::snapshot(_instructions_retired_counter);
204  }
205 
206  [[gnu::always_inline]] [[gnu::hot]]
207  void stop_iteration() {
208  auto t = clock_type::now();
209  _total_time += t - _start_time;
210  perf_stats stats;
211  stats = perf_stats::snapshot(_instructions_retired_counter);
212  _total_stats += stats - _start_stats;
213  }
214 };

243 template<typename Test>
244 class concrete_performance_test final : public performance_test {
245  std::optional<Test> _test;
246 private:
247  template<typename... Args>
248  auto run_test(Args&&...) {
249  return _test->run();
250  }
251 
252 protected:
253  virtual void set_up() override {
254  _test.emplace();
255  }
256 
257  virtual void tear_down() noexcept override {
258  _test = std::nullopt;
259  }
260 
261  [[gnu::hot]]
262  virtual future<run_result> do_single_run() override {
263  // Redundant 'this->'s courtesy of https://gcc.gnu.org/bugzilla/show_bug.cgi?id=61636
264  _instructions_retired_counter.enable();
265  return if_constexpr_<is_future<decltype(_test->run())>::value>([&] (auto&&...) {
266  measure_time.start_run(&_instructions_retired_counter);
267  return do_until([this] { return this->stop_iteration(); }, [this] {
268  return if_constexpr_<std::is_same_v<decltype(_test->run()), future<>>>([&] (auto&&...) {
269  this->next_iteration(1);
270  return _test->run();
271  }, [&] (auto&&... dependency) {
272  // We need `dependency` to make sure the compiler won't be able to instantiate anything
273  // (and notice that the code does not compile) if this part of if_constexpr_ is not active.
274  return run_test(dependency...).then([&] (size_t n) {
275  this->next_iteration(n);
276  });
277  })();
278  }).then([] {
279  return measure_time.stop_run();
280  }).finally([this] {
281  _instructions_retired_counter.disable();
282  });
283  }, [&] (auto&&...) {
284  measure_time.start_run(&_instructions_retired_counter);
285  while (!stop_iteration()) {
286  if_constexpr_<std::is_void_v<decltype(_test->run())>>([&] (auto&&...) {
287  (void)_test->run();
288  this->next_iteration(1);
289  }, [&] (auto&&... dependency) {
290  // We need `dependency` to make sure the compiler won't be able to instantiate anything
291  // (and notice that the code does not compile) if this part of if_constexpr_ is not active.
292  this->next_iteration(run_test(dependency...));
293  })();
294  }
295  auto ret = measure_time.stop_run();
296  _instructions_retired_counter.disable();
297  return make_ready_future<run_result>(std::move(ret));
298  })();
299  }
300 public:
301  using performance_test::performance_test;
302 };

337 // PERF_TEST and PERF_TEST_F support both synchronous and asynchronous functions.
338 // The former should return `void`, the latter `future<>`.
339 // PERF_TEST_C executes a coroutine function, if enabled.
340 // PERF_TEST_CN executes a coroutine function, if enabled, returning the number of inner-loops.
341 //
342 // Test cases may perform multiple operations in a single run, this may be desirable
343 // if the cost of an individual operation is very small. This allows measuring either
344 // the latency of throughput depending on how the test in written. In such cases,
345 // the test function shall return either size_t or future<size_t> for synchronous and
346 // asynchronous cases respectively. The returned value shall be the number of iterations
347 // done in a single test run.
348 
349 #define PERF_TEST_F(test_group, test_case) \
350  struct test_##test_group##_##test_case : test_group { \
351  [[gnu::always_inline]] inline auto run(); \
352  }; \
353  static ::perf_tests::internal::test_registrar<test_##test_group##_##test_case> \
354  test_##test_group##_##test_case##_registrar(#test_group, #test_case); \
355  [[gnu::always_inline]] auto test_##test_group##_##test_case::run()
356 
357 #define PERF_TEST(test_group, test_case) \
