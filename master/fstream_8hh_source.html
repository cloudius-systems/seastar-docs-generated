93 struct file_output_stream_options {
94  // For small files, setting preallocation_size can make it impossible for XFS to find
95  // an aligned extent. On the other hand, without it, XFS will divide the file into
96  // file_size/buffer_size extents. To avoid fragmentation, we set the default buffer_size
97  // to 64k (so each extent will be a minimum of 64k) and preallocation_size to 0 (to avoid
98  // extent allocation problems).
99  //
