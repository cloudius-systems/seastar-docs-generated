72  void* future;

75 class when_all_state_base {
76  size_t _nr_remain;
77  const when_all_process_element* _processors;
78  void* _continuation;
79 public:
80  virtual ~when_all_state_base() {}
81  when_all_state_base(size_t nr_remain, const when_all_process_element* processors, void* continuation)
82  : _nr_remain(nr_remain), _processors(processors), _continuation(continuation) {
83  }
84  virtual task* waiting_task() = 0;
85  void complete_one() noexcept {
86  // We complete in reverse order; if the futures happen to complete
87  // in order, then waiting for the last one will find the rest ready
88  --_nr_remain;
89  while (_nr_remain) {
90  bool ready = process_one(_nr_remain - 1);
91  if (!ready) {
92  return;
93  }
94  --_nr_remain;
95  }
96  if (!_nr_remain) {
97  delete this;
98  }
99  }
100  void do_wait_all() noexcept {
101  ++_nr_remain; // fake pending completion for complete_one()
102  complete_one();
103  }
104  bool process_one(size_t idx) noexcept {
105  auto p = _processors[idx];
106  return p.func(p.future, _continuation, this);
107  }
108 };

110 template <typename Future>
111 class when_all_state_component final : public continuation_base_from_future_t<Future> {
112  when_all_state_base* _base;
113  Future* _final_resting_place;
114 public:
115  static bool process_element_func(void* future, void* continuation, when_all_state_base* wasb) noexcept {
116  auto f = reinterpret_cast<Future*>(future);
117  if (f->available()) {
118  return true;
119  } else {
120  auto c = new (continuation) when_all_state_component(wasb, f);
121  set_callback(std::move(*f), c);
122  return false;
123  }
124  }
125  when_all_state_component(when_all_state_base *base, Future* future) noexcept : _base(base), _final_resting_place(future) {}
126  task* waiting_task() noexcept override { return _base->waiting_task(); }
127  virtual void run_and_dispose() noexcept override {
128  using futurator = futurize<Future>;
129  if (__builtin_expect(this->_state.failed(), false)) {
130  *_final_resting_place = futurator::make_exception_future(std::move(this->_state).get_exception());
131  } else {
132  *_final_resting_place = futurator::from_tuple(std::move(this->_state).get_value());
133  }
134  auto base = _base;
135  this->~when_all_state_component();
136  base->complete_one();
137  }
138 };

140 template<typename ResolvedTupleTransform, typename... Futures>
141 class when_all_state : public when_all_state_base {
142  static constexpr size_t nr = sizeof...(Futures);
143  using type = std::tuple<Futures...>;
144  type tuple;
145  // We only schedule one continuation at a time, and store it in _cont.
146  // This way, if while the future we wait for completes, some other futures
147  // also complete, we won't need to schedule continuations for them.
148  std::aligned_union_t<1, when_all_state_component<Futures>...> _cont;
149  when_all_process_element _processors[nr];
150 public:
151  typename ResolvedTupleTransform::promise_type p;
152  when_all_state(Futures&&... t) : when_all_state_base(nr, _processors, &_cont), tuple(std::make_tuple(std::move(t)...)) {
153  init_element_processors(std::make_index_sequence<nr>());
154  }
155  virtual ~when_all_state() {
156  ResolvedTupleTransform::set_promise(p, std::move(tuple));
157  }
158  task* waiting_task() noexcept override {
159  return p.waiting_task();
160  }
161 private:
162  template <size_t... Idx>
163  void init_element_processors(std::index_sequence<Idx...>) {
164  auto ignore = {
165  0,
166  (_processors[Idx] = when_all_process_element{
167  when_all_state_component<std::tuple_element_t<Idx, type>>::process_element_func,
168  &std::get<Idx>(tuple)
169  }, 0)...
170  };
171  (void)ignore;
172  }
173 public:
174  static typename ResolvedTupleTransform::future_type wait_all(Futures&&... futures) noexcept {
175  if ((futures.available() && ...)) {
176  return ResolvedTupleTransform::make_ready_future(std::make_tuple(std::move(futures)...));
177  }
178  auto state = [&] () noexcept {
179  memory::scoped_critical_alloc_section _;
180  return new when_all_state(std::move(futures)...);
181  }();
182  auto ret = state->p.get_future();
183  state->do_wait_all();
184  return ret;
185  }
186 };

272 // Internal function for when_all().
273 template <typename ResolvedVectorTransform, typename Future>
274 inline
275 typename ResolvedVectorTransform::future_type
276 complete_when_all(std::vector<Future>&& futures, typename std::vector<Future>::iterator pos) noexcept {
277  // If any futures are already ready, skip them.
278  while (pos != futures.end() && pos->available()) {
279  ++pos;
280  }
281  // Done?
282  if (pos == futures.end()) {
283  return ResolvedVectorTransform::run(std::move(futures));
284  }
285  // Wait for unready future, store, and continue.
286  return pos->then_wrapped([futures = std::move(futures), pos] (auto fut) mutable {
287  *pos++ = std::move(fut);
288  return complete_when_all<ResolvedVectorTransform>(std::move(futures), pos);
289  });
290 }

292 template<typename ResolvedVectorTransform, typename FutureIterator>
293 inline auto
294 do_when_all(FutureIterator begin, FutureIterator end) noexcept {
295  using itraits = std::iterator_traits<FutureIterator>;
296  auto make_values_vector = [] (size_t size) noexcept {
297  memory::scoped_critical_alloc_section _;
298  std::vector<typename itraits::value_type> ret;
299  ret.reserve(size);
300  return ret;
301  };
302  std::vector<typename itraits::value_type> ret =
303  make_values_vector(iterator_range_estimate_vector_capacity(begin, end, typename itraits::iterator_category()));
304  // Important to invoke the *begin here, in case it's a function iterator,
305  // so we launch all computation in parallel.
306  std::move(begin, end, std::back_inserter(ret));
307  return complete_when_all<ResolvedVectorTransform>(std::move(ret), ret.begin());
308 }

368 template<typename... Futures>
369 class extract_values_from_futures_tuple {
370  static auto transform(std::tuple<Futures...> futures) noexcept {
371  auto prepare_result = [] (auto futures) noexcept {
372  auto fs = tuple_filter_by_type<internal::future_has_value>(std::move(futures));
373  return tuple_map(std::move(fs), [] (auto&& e) {
374  return e.get0();
375  });
376  };
377 
378  using tuple_futurizer = internal::tuple_to_future<decltype(prepare_result(std::move(futures)))>;
379 
380  std::exception_ptr excp;
381  tuple_for_each(futures, [&excp] (auto& f) {
382  if (!excp) {
383  if (f.failed()) {
384  excp = f.get_exception();
385  }
386  } else {
387  f.ignore_ready_future();
388  }
389  });
390  if (excp) {
391  return tuple_futurizer::make_failed(std::move(excp));
392  }
393 
394  return tuple_futurizer::make_ready(prepare_result(std::move(futures)));
395  }
396 public:
397  using future_type = decltype(transform(std::declval<std::tuple<Futures...>>()));
398  using promise_type = typename future_type::promise_type;
399 
400  static void set_promise(promise_type& p, std::tuple<Futures...> tuple) {
401  transform(std::move(tuple)).forward_to(std::move(p));
402  }
403 
404  static future_type make_ready_future(std::tuple<Futures...> tuple) noexcept {
405  return transform(std::move(tuple));
406  }
407 
408  static future_type current_exception_as_future() noexcept {
409  future_type (*type_deduct)() = current_exception_as_future;
410  return type_deduct();
411  }
412 };

414 template<typename Future>
415 struct extract_values_from_futures_vector {
416  using value_type = decltype(untuple(std::declval<typename Future::tuple_type>()));
417 
418  using future_type = future<std::vector<value_type>>;
419 
420  static future_type run(std::vector<Future> futures) noexcept {
421  auto make_values_vector = [] (size_t size) noexcept {
422  memory::scoped_critical_alloc_section _;
423  std::vector<value_type> values;
424  values.reserve(size);
425  return values;
426  };
427  std::vector<value_type> values = make_values_vector(futures.size());
428 
429  std::exception_ptr excp;
430  for (auto&& f : futures) {
431  if (!excp) {
432  if (f.failed()) {
433  excp = f.get_exception();
434  } else {
435  values.emplace_back(f.get0());
436  }
437  } else {
438  f.ignore_ready_future();
439  }
440  }
441  if (excp) {
442  return seastar::make_exception_future<std::vector<value_type>>(std::move(excp));
443  }
444  return make_ready_future<std::vector<value_type>>(std::move(values));
445  }
446 
447  static future_type current_exception_as_future() noexcept {
448  return seastar::current_exception_as_future<std::vector<value_type>>();
449  }
