93 template <typename T, typename Period, capped_release Capped, typename Clock = std::chrono::steady_clock>
94 SEASTAR_CONCEPT( requires std::is_nothrow_copy_constructible_v<T> && supports_wrapping_arithmetics<T> )
95 class shared_token_bucket {
96  using rate_resolution = std::chrono::duration<double, Period>;
97 
98  T _replenish_rate;
99  const T _replenish_limit;
100  const T _replenish_threshold;
101  std::atomic<typename Clock::time_point> _replenished;
102 
103  /*
104  * The token bucket is implemented as a pair of wrapping monotonic
105  * counters (called rovers) one chasing the other. Getting a token
106  * from the bucket is increasing the tail, replenishing a token back
107  * is increasing the head. If increased tail overruns the head then
108  * the bucket is empty and we have to wait. The shard that grabs tail
109  * earlier will be "woken up" earlier, so they form a queue.
110  *
111  * The top rover is needed to implement two buckets actually. The
112  * tokens are not just replenished by timer. They are replenished by
113  * timer from the second bucket. And the second bucket only get a
114  * token in it after the request that grabbed it from the first bucket
115  * completes and returns it back.
116  */
117 
118  using rovers_t = rovers<T, Capped>;
119  static_assert(rovers_t::atomic_rover::is_always_lock_free);
120  rovers_t _rovers;
121 
122  T tail() const noexcept { return _rovers.tail.load(std::memory_order_relaxed); }
123  T head() const noexcept { return _rovers.head.load(std::memory_order_relaxed); }
124 
125  /*
126  * Need to make sure that the multiplication in accumulated_in() doesn't
127  * overflow. Not to introduce an extra branch there, define that the
128  * replenish period is not larger than this delta and limit the rate with
129  * the value that can overflow it.
130  *
131  * The additional /=2 in max_rate math is to make extra sure that the
132  * overflow doesn't break wrapping_difference sign tricks.
133  */
134  static constexpr rate_resolution max_delta = std::chrono::duration_cast<rate_resolution>(std::chrono::hours(1));
135 public:
136  static constexpr T max_rate = std::numeric_limits<T>::max() / 2 / max_delta.count();
137  static constexpr capped_release is_capped = Capped;
138 
139 private:
140  static constexpr T accumulated(T rate, rate_resolution delta) noexcept {
141  return std::round(rate * delta.count());
142  }
143 #ifndef __clang__
144  // std::round() is constexpr only since C++23 (but g++ doesn't care)
145  static_assert(accumulated(max_rate, max_delta) <= std::numeric_limits<T>::max());
146 #endif
147 
148 public:
149  shared_token_bucket(T rate, T limit, T threshold, bool add_replenish_iffset = true) noexcept
150  : _replenish_rate(std::min(rate, max_rate))
151  , _replenish_limit(limit)
152  , _replenish_threshold(std::clamp(threshold, (T)1, limit))
153  // pretend it was replenished yesterday to spot overflows early
154  , _replenished(Clock::now() - std::chrono::hours(add_replenish_iffset ? 24 : 0))
155  , _rovers(_replenish_limit)
156  {}
157 
158  T grab(T tokens) noexcept {
159  return fetch_add(_rovers.tail, tokens) + tokens;
160  }
161 
162  void release(T tokens) noexcept {
163  _rovers.release(tokens);
164  }
165 
166  void replenish(typename Clock::time_point now) noexcept {
167  auto ts = _replenished.load(std::memory_order_relaxed);
168 
169  if (now <= ts) {
170  return;
171  }
172 
173  auto delta = now - ts;
174  auto extra = accumulated_in(delta);
175 
176  if (extra >= _replenish_threshold) {
177  if (!_replenished.compare_exchange_weak(ts, ts + delta)) {
178  return; // next time or another shard
179  }
180 
181  fetch_add(_rovers.head, std::min(extra, _rovers.max_extra(_replenish_limit)));
182  }
183  }
184 
185  T deficiency(T from) const noexcept {
186  return wrapping_difference(from, head());
187  }
188 
189  template <typename Rep, typename Per>
190  static auto rate_cast(const std::chrono::duration<Rep, Per> delta) noexcept {
191  return std::chrono::duration_cast<rate_resolution>(delta);
192  }
193 
194  // the number of tokens accumulated for the given time frame
195  template <typename Rep, typename Per>
196  T accumulated_in(const std::chrono::duration<Rep, Per> delta) const noexcept {
197  auto delta_at_rate = std::min(rate_cast(delta), max_delta);
198  return accumulated(_replenish_rate, delta_at_rate);
199  }
200 
201  // Estimated time to process the given amount of tokens
202  // (peer of accumulated_in helper)
203  rate_resolution duration_for(T tokens) const noexcept {
204  return rate_resolution(double(tokens) / _replenish_rate);
205  }
206 
207  T rate() const noexcept { return _replenish_rate; }
208  T limit() const noexcept { return _replenish_limit; }
209  T threshold() const noexcept { return _replenish_threshold; }
210  typename Clock::time_point replenished_ts() const noexcept { return _replenished; }
211 
212  void update_rate(T rate) noexcept {
213  _replenish_rate = std::min(rate, max_rate);
214  }
215 };
