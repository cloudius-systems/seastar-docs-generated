108 enum class file_permissions {

110  user_write = S_IWUSR, // Write by owner
111  user_execute = S_IXUSR, // Execute by owner
112 
113  group_read = S_IRGRP, // Read by group
114  group_write = S_IWGRP, // Write by group
115  group_execute = S_IXGRP, // Execute by group
116 
117  others_read = S_IROTH, // Read by others
118  others_write = S_IWOTH, // Write by others
119  others_execute = S_IXOTH, // Execute by others
120 
121  user_permissions = user_read | user_write | user_execute,
122  group_permissions = group_read | group_write | group_execute,
123  others_permissions = others_read | others_write | others_execute,
124  all_permissions = user_permissions | group_permissions | others_permissions,
125 
126  default_file_permissions = user_read | user_write | group_read | group_write | others_read | others_write, // 0666
127  default_dir_permissions = all_permissions, // 0777
128 };
