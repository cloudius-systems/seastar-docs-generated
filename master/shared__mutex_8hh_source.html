49 class shared_mutex {
50  unsigned _readers = 0;
51  bool _writer = false;
52  struct waiter {
53  waiter(promise<>&& pr, bool for_write) : pr(std::move(pr)), for_write(for_write) {}
54  promise<> pr;
55  bool for_write;
56  };
57  circular_buffer<waiter> _waiters;
58 public:
59  shared_mutex() = default;
60  shared_mutex(shared_mutex&&) = default;
61  shared_mutex& operator=(shared_mutex&&) = default;
62  shared_mutex(const shared_mutex&) = delete;
63  void operator=(const shared_mutex&) = delete;
68  future<> lock_shared() {
69  if (try_lock_shared()) {
70  return make_ready_future<>();
71  }
72  _waiters.emplace_back(promise<>(), false);
73  return _waiters.back().pr.get_future();
74  }
78  bool try_lock_shared() noexcept {
79  if (!_writer && _waiters.empty()) {
80  ++_readers;
81  return true;
82  }
83  return false;
84  }
86  void unlock_shared() {
87  assert(_readers > 0);
88  --_readers;
89  wake();
90  }
95  future<> lock() {
96  if (try_lock()) {
97  return make_ready_future<>();
98  }
99  _waiters.emplace_back(promise<>(), true);
100  return _waiters.back().pr.get_future();
101  }
105  bool try_lock() noexcept {
106  if (!_readers && !_writer) {
107  _writer = true;
108  return true;
109  }
110  return false;
111  }
113  void unlock() {
114  assert(_writer);
115  _writer = false;
116  wake();
117  }
118 private:
119  void wake() {
120  while (!_waiters.empty()) {
121  auto& w = _waiters.front();
122  // note: _writer == false in wake()
123  if (w.for_write) {
124  if (!_readers) {
125  _writer = true;
126  w.pr.set_value();
127  _waiters.pop_front();
128  }
129  break;
130  } else { // for read
131  ++_readers;
132  w.pr.set_value();
133  _waiters.pop_front();
134  }
135  }
136  }
137 };
